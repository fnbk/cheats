#
# patch
#

# create patch
svn diff TSES/ > STANDALONE.patch

# apply patch (-p0: explicitly tell to strip 0 components from file names)
patch -p0 -i STANDALONE.patch

# undo patch
patch -R -p0 -i STANDALONE.patch


#

#
